<#
.SYNOPSIS
    Reads a Class Completion Data .xls/.xlsx export, sorts it by
    Day of Week -> Room Name -> Period Name -> Period Start Time,
    and generates a styled, self-contained HTML report.

.DESCRIPTION
    Works on any Windows PC that has Microsoft Excel installed (used only
    to read the legacy binary .xls). No extra modules required.

.PARAMETER InputPath
    One or more source spreadsheets (.xls or .xlsx). Accepts a comma-separated
    list, an array, or wildcards. When multiple files are given, all of their
    rows are merged into a single report (use the Unit filter to separate them).
    Defaults to every "Class_Data_*.xls*" file in your Downloads folder.

.PARAMETER OutputPath
    Path for the generated HTML report.
    Defaults to "<name>_Sorted.html" for a single file, or
    "Class_Report_Combined_<date>.html" when multiple files are merged.

.EXAMPLE
    .\Generate-ClassReport.ps1

.EXAMPLE
    .\Generate-ClassReport.ps1 -InputPath "C:\data\Class_Data.xlsx" -OutputPath "C:\data\report.html"

.EXAMPLE
    # Merge several unit exports into one report
    .\Generate-ClassReport.ps1 -InputPath "C:\data\Troop962.xlsx","C:\data\Troop17.xlsx","C:\data\Pack5.xlsx"

.EXAMPLE
    # Merge everything matching a wildcard
    .\Generate-ClassReport.ps1 -InputPath "C:\data\*.xlsx"
#>

[CmdletBinding()]
param(
    [string[]]$InputPath,
    [string]$OutputPath
)

$ErrorActionPreference = 'Stop'

# ---- Resolve input files --------------------------------------------------
if (-not $InputPath -or $InputPath.Count -eq 0) {
    $dl = Join-Path $env:USERPROFILE 'Downloads'
    $candidates = Get-ChildItem -Path $dl -Filter 'Class_Data_*.xls*' -ErrorAction SilentlyContinue |
                  Sort-Object LastWriteTime -Descending
    if (-not $candidates) {
        throw "No input file given and no 'Class_Data_*.xls*' found in $dl. Use -InputPath."
    }
    $InputPath = $candidates.FullName
}

# Expand wildcards and de-duplicate, preserving order.
# Also split any comma-joined element so both `-InputPath a,b` (in-session
# array) and `powershell -File ... -InputPath "a,b"` styles work.
$inputFiles = @()
$patterns = @()
foreach ($item in $InputPath) {
    foreach ($piece in ($item -split ',')) {
        $piece = $piece.Trim().Trim('"')
        if ($piece) { $patterns += $piece }
    }
}
foreach ($pattern in $patterns) {
    $resolved = Resolve-Path -Path $pattern -ErrorAction SilentlyContinue
    if (-not $resolved) { throw "Input file not found: $pattern" }
    foreach ($rp in $resolved) {
        if ($inputFiles -notcontains $rp.Path) { $inputFiles += $rp.Path }
    }
}
if ($inputFiles.Count -eq 0) { throw "No input files resolved from -InputPath." }

if (-not $OutputPath) {
    $dir = Split-Path $inputFiles[0] -Parent
    if ($inputFiles.Count -eq 1) {
        $name = [System.IO.Path]::GetFileNameWithoutExtension($inputFiles[0])
        $OutputPath = Join-Path $dir ($name + '_Sorted.html')
    } else {
        $OutputPath = Join-Path $dir ('Class_Report_Combined_' + (Get-Date -Format 'yyyy_MM_dd') + '.html')
    }
}

Write-Host ("Reading : {0} file(s)" -f $inputFiles.Count)

# ---- Read the workbook(s) via Excel COM -----------------------------------
$excel = $null
$dataRows = @()
$headers  = @()   # canonical headers, taken from the first file
try {
    $excel = New-Object -ComObject Excel.Application
    $excel.Visible = $false
    $excel.DisplayAlerts = $false

    foreach ($file in $inputFiles) {
        Write-Host "          - $(Split-Path $file -Leaf)"
        $wb = $excel.Workbooks.Open($file, [Type]::Missing, $true) # read-only
        try {
            $ws = $wb.Sheets.Item(1)
            $used = $ws.UsedRange
            $rowCount = $used.Rows.Count
            $colCount = $used.Columns.Count
            $grid = $used.Value2

            # Read this file's header row.
            $fileHeaders = @()
            for ($c = 1; $c -le $colCount; $c++) {
                $h = $grid[1, $c]
                if ($null -eq $h) { $h = "Col$c" }
                $fileHeaders += [string]$h.ToString().Trim()
            }

            if ($headers.Count -eq 0) { $headers = $fileHeaders }

            # Map each canonical header to a column index in THIS file
            # (exact match first, then prefix match to tolerate truncation).
            $colMap = @{}
            foreach ($cn in $headers) {
                $idx = [array]::IndexOf($fileHeaders, $cn)
                if ($idx -lt 0) {
                    for ($k = 0; $k -lt $fileHeaders.Count; $k++) {
                        if ($fileHeaders[$k] -like ($cn + '*') -or $cn -like ($fileHeaders[$k] + '*')) { $idx = $k; break }
                    }
                }
                $colMap[$cn] = $idx  # -1 if absent
            }

            for ($r = 2; $r -le $rowCount; $r++) {
                $obj = [ordered]@{}
                foreach ($cn in $headers) {
                    $idx = $colMap[$cn]
                    $v = if ($idx -ge 0) { $grid[$r, ($idx + 1)] } else { $null }
                    $obj[$cn] = if ($null -eq $v) { '' } else { [string]$v.ToString().Trim() }
                }
                $row = [pscustomobject]$obj
                $row | Add-Member -NotePropertyName '_Source' -NotePropertyValue (Split-Path $file -Leaf)
                $dataRows += $row
            }
        }
        finally {
            if ($wb) { $wb.Close($false) | Out-Null }
            if ($ws) { [System.Runtime.InteropServices.Marshal]::ReleaseComObject($ws) | Out-Null }
            if ($wb) { [System.Runtime.InteropServices.Marshal]::ReleaseComObject($wb) | Out-Null }
            $ws = $null; $wb = $null
        }
    }
}
finally {
    if ($excel) { $excel.Quit() }
    if ($excel) { [System.Runtime.InteropServices.Marshal]::ReleaseComObject($excel) | Out-Null }
    [GC]::Collect(); [GC]::WaitForPendingFinalizers()
}

Write-Host ("Loaded  : {0} rows from {1} file(s), {2} columns" -f $dataRows.Count, $inputFiles.Count, $headers.Count)

# ---- Helpers --------------------------------------------------------------

# Day columns (export truncates some names). Map header -> order + clean label.
$dayDefs = @(
    @{ Order = 0; Label = 'Sunday';    Match = 'Sunday'    },
    @{ Order = 1; Label = 'Monday';    Match = 'Monday'    },
    @{ Order = 2; Label = 'Tuesday';   Match = 'Tuesday'   },
    @{ Order = 3; Label = 'Wednesday'; Match = 'Wednesda'  },
    @{ Order = 4; Label = 'Thursday';  Match = 'Thursda'   },
    @{ Order = 5; Label = 'Friday';    Match = 'Friday'    },
    @{ Order = 6; Label = 'Saturday';  Match = 'Saturda'   }
)

# Resolve the actual day-column header names once (handles truncation).
$dayColMap = foreach ($d in $dayDefs) {
    $col = $headers | Where-Object { $_ -like ($d.Match + '*') } | Select-Object -First 1
    [pscustomobject]@{ Order = $d.Order; Label = $d.Label; Col = $col }
}

# Returns EVERY day marked with an "X" for a row (multi-day courses mark more
# than one). Falls back to a single "(No Day)" entry when none are marked.
function Get-DayList($row) {
    $hits = @()
    foreach ($d in $dayColMap) {
        if ($d.Col -and ("$($row.$($d.Col))".Trim().ToUpper() -eq 'X')) {
            $hits += [pscustomobject]@{ Order = $d.Order; Label = $d.Label }
        }
    }
    if ($hits.Count -eq 0) { $hits = ,([pscustomobject]@{ Order = 99; Label = '(No Day)' }) }
    return $hits
}

function Get-TimeSortKey([string]$t) {
    if ([string]::IsNullOrWhiteSpace($t)) { return [double]99999 }
    $dt = [datetime]::MinValue
    if ([datetime]::TryParse($t, [ref]$dt)) { return [double]($dt.TimeOfDay.TotalMinutes) }
    return [double]99998
}

function HtmlEncode([string]$s) {
    if ($null -eq $s) { return '' }
    return [System.Web.HttpUtility]::HtmlEncode($s)
}
Add-Type -AssemblyName System.Web

# Resolve actual header names we display (handles trailing-space quirks).
function ColName($pattern) { ($headers | Where-Object { $_ -like $pattern } | Select-Object -First 1) }
$cRoom   = ColName 'Room Name*'
$cPeriod = ColName 'Period Name*'
$cStart  = ColName 'Period Start Time*'
$cEnd    = ColName 'Class End Time*'
$cTitle  = ColName 'Class Title*'
$cNum    = ColName 'Class Number*'
$cInstr  = ColName 'Instructor Name*'
$cAtt    = ColName 'Attendee Name*'
$cUnit   = ColName 'Unit*'

# ---- Fix provider data error: full-day classes mislabeled as Morning ------
# Some rows are tagged "Morning Session" but run 8:30 AM -> 4:30 PM, i.e. they
# span both the morning AND afternoon sessions, yet no matching afternoon row
# exists. For each such row we add a duplicate that appears as a proper
# "Afternoon Session" entry so it shows up in the afternoon as well.

# Canonical afternoon start time, derived from the data (fallback 1:30 PM).
$afternoonStart = ($dataRows |
    Where-Object { "$($_.$cPeriod)" -like 'Afternoon*' -and "$($_.$cStart)".Trim() } |
    Group-Object { "$($_.$cStart)".Trim() } |
    Sort-Object Count -Descending | Select-Object -First 1).Name
if (-not $afternoonStart) { $afternoonStart = '1:30 PM' }
$afternoonStartMin = Get-TimeSortKey $afternoonStart

$synthRows = @()
foreach ($row in $dataRows) {
    if ("$($row.$cPeriod)" -notlike 'Morning*') { continue }
    $endMin = Get-TimeSortKey "$($row.$cEnd)"
    # End time runs into the afternoon -> this is really a full-day class.
    if ($endMin -lt 90000 -and $endMin -gt $afternoonStartMin) {
        $clone = [ordered]@{}
        foreach ($prop in $row.PSObject.Properties) { $clone[$prop.Name] = $prop.Value }
        $clone[$cPeriod] = 'Afternoon Session'
        $clone[$cStart]  = $afternoonStart
        $cloneObj = [pscustomobject]$clone
        $cloneObj | Add-Member -NotePropertyName '_Synth' -NotePropertyValue $true
        $synthRows += $cloneObj
    }
}
if ($synthRows.Count -gt 0) {
    $dataRows += $synthRows
    Write-Host ("Fixed   : added {0} afternoon duplicate(s) for full-day Morning classes" -f $synthRows.Count)
}

# ---- Sort -----------------------------------------------------------------
# Expand each row into one entry per marked day (multi-day courses appear on
# every day they run).
$enriched = foreach ($row in $dataRows) {
    foreach ($di in (Get-DayList $row)) {
        [pscustomobject]@{
            Row       = $row
            DayOrder  = $di.Order
            DayLabel  = $di.Label
            RoomKey   = "$($row.$cRoom)"
            PeriodKey = "$($row.$cPeriod)"
            TitleKey  = "$($row.$cTitle)"
            TimeKey   = Get-TimeSortKey "$($row.$cStart)"
            IsSynth   = [bool]$row.'_Synth'
        }
    }
}

# Sort: Day -> Start Time -> Room -> Class Title
$sorted = $enriched | Sort-Object DayOrder, TimeKey, RoomKey, TitleKey

# ---- Build HTML -----------------------------------------------------------
$generated = Get-Date -Format 'dddd, MMMM d, yyyy h:mm tt'
$srcLeaves = $inputFiles | ForEach-Object { Split-Path $_ -Leaf }
if ($srcLeaves.Count -eq 1) { $srcName = $srcLeaves[0] }
else { $srcName = "$($srcLeaves.Count) files: " + ($srcLeaves -join ', ') }

$grouped = $sorted | Group-Object DayLabel
